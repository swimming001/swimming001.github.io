主席树的血域泪

引言：为什么叫“血与泪”？因为哪怕少开大一倍数组，或者离散化时数错了一个下标，迎接你的都是无情的 RE 或 WA。今天用最纯粹的代码，拆解可持久化线段树（主席树）的优雅之处。

🛠️ 核心原理：我们在解决什么问题？

主席树的核心任务是：求区间第 $k$ 大/小。
如果只是求全局第 $k$ 小，我们只需要一棵线段树维护值域。但如果要查区间 $[L, R]$，利用前缀和的思想，我们需要第 $R$ 步的线段树减去第 $L-1$ 步的线段树。

如果每一步都建一棵新树，空间复杂度是 $\mathcal{O}(N^2)$，直接爆炸。
所以我们要**“偷懒”**——只新建有变化的节点，剩下的全部连向历史版本。

💻 核心代码与精妙之处拆解

下面是完整的实现代码，里面藏着几个非常高级的技巧。

#include<bits/stdc++.h>
using namespace std;
using ll=long long;

int cnt=1;
struct Node {
    int val;
    int l,r;
}a[5000000]; // 🩸 血泪点1：数组一定要开够大！通常是 N * 40

// ✨ 精妙点1：完全不需要 build 方法
int insert(int q, int i, int l, int r) {
    int p = cnt++;
    a[p] = a[q]; // ✨ 精妙点2：直接复制历史节点的所有信息（左右儿子指针和权值）
    a[p].val++;
    if (l == r) { return p; }
    int mid = (l + r) >> 1;
    if (i <= mid) a[p].l = insert(a[q].l, i, l, mid);
    if (i > mid)  a[p].r = insert(a[q].r, i, mid + 1, r);
    return p;
}

int query(int q, int p, int k, int l, int r) {
    if (l == r) { return l; } // 返回离散化后的下标
    int dif = a[a[q].l].val - a[a[p].l].val; // 前缀和思想：当前区间左子树贡献的频次
    int mid = (l + r) >> 1;
    if (k <= dif) { return query(a[q].l, a[p].l, k, l, mid); }
    return query(a[q].r, a[p].r, k - dif, mid + 1, r); // 去右子树找第 k - dif 小
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int n, q; cin >> n >> q;
    vector<ll> nums(n + 1);
    vector<ll> nums1(n + 1); // 副本，用于离散化
    unordered_map<ll, int> mp;
    for (int i = 1; i <= n; i++) {
        cin >> nums[i];
        nums1[i] = nums[i];
    }
    
    // 🩸 血泪点2：离散化三步走（注意 vector 1-indexed 的擦除问题）
    sort(nums1.begin() + 1, nums1.end());
    auto it = unique(nums1.begin() + 1, nums1.end());
    nums1.erase(it, nums1.end());
    int s = nums1.size() - 1;
    for (int i = 1; i <= s; i++) { mp[nums1[i]] = i; }
    
    // 根节点数组，用于记录每次插入后的新树根
    vector<int> root(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        root[i] = insert(root[i - 1], mp[nums[i]], 1, s); // 复用上一棵树 root[i-1]
    }
    
    // 询问
    for (int i = 0; i < q; i++) {
        int u, v, k; cin >> u >> v >> k;
        // 🩸 血泪点3：前缀和区间是 root[v] 和 root[u-1]
        cout << nums1[query(root[v], root[u - 1], k, 1, s)] << '\n';
    }
    return 0;
}
